create table announcements
(
    id                   bigint auto_increment comment '公告ID'
        primary key,
    publisher            varchar(50)                          null comment '发布者ID，逻辑关联 users.id',
    title                varchar(255)                         not null comment '公告标题',
    content_json         json                                 null comment '富文本JSON内容',
    content_md           longtext                             null comment 'Markdown内容',
    image_url            text                                 null comment '公告图片URL',
    created_at           datetime   default CURRENT_TIMESTAMP not null comment '创建时间',
    last_modified_at     datetime   default CURRENT_TIMESTAMP not null on update CURRENT_TIMESTAMP comment '更新时间',
    is_pinned_by_manager tinyint(1) default 0                 not null comment '是否管理员置顶',
    num_likes            int        default 0                 not null comment '点赞数',
    num_reads            int        default 0                 not null comment '阅读数'
)
    comment '公告表';

create index idx_announcements_created
    on announcements (created_at);

create index idx_announcements_pinned
    on announcements (is_pinned_by_manager);

create index idx_announcements_publisher
    on announcements (publisher);

create table comment_likes
(
    id         bigint auto_increment comment '评论点赞ID'
        primary key,
    user_id    bigint                             not null comment '用户ID',
    comment_id bigint                             not null comment '评论ID',
    created_at datetime default CURRENT_TIMESTAMP not null comment '点赞时间',
    constraint uk_comment_like
        unique (user_id, comment_id)
)
    comment '评论点赞表';

create index idx_comment_likes_comment
    on comment_likes (comment_id);

create index idx_comment_likes_user
    on comment_likes (user_id);

create table comments
(
    comment_id   bigint auto_increment comment '评论ID'
        primary key,
    post_id      bigint                                not null comment '帖子ID，逻辑关联 posts.id',
    root_id      bigint                                null comment '根评论ID',
    user_id      bigint                                not null comment '评论用户ID，逻辑关联 users.id',
    username     varchar(50)                           not null comment '用户名',
    created_at   datetime    default CURRENT_TIMESTAMP not null comment '创建时间',
    content      text                                  not null comment '评论内容',
    status       varchar(20) default 'normal'          not null comment '状态：normal/deleted/hidden',
    likes        int         default 0                 not null comment '评论点赞数',
    level        int         default 0                 not null comment '评论层级',
    has_children tinyint(1)  default 0                 null comment '是否有子评论',
    avatar_url   varchar(255)                          null comment '用户头像',
    updated_at   datetime    default CURRENT_TIMESTAMP not null on update CURRENT_TIMESTAMP comment '更新时间'
)
    comment '评论表';

create index idx_comments_post
    on comments (post_id);

create index idx_comments_post_parent
    on comments (post_id);

create index idx_comments_root
    on comments (root_id);

create index idx_comments_user
    on comments (user_id);

create table follows
(
    id           bigint auto_increment comment '关注关系ID'
        primary key,
    follower_id  bigint                             not null comment '关注者用户ID',
    following_id bigint                             not null comment '被关注者用户ID',
    created_at   datetime default CURRENT_TIMESTAMP not null comment '关注时间',
    constraint uk_follow_pair
        unique (follower_id, following_id)
)
    comment '用户关注关系表';

create index idx_follows_follower
    on follows (follower_id);

create index idx_follows_following
    on follows (following_id);

create table post_reads
(
    id           bigint auto_increment comment '阅读记录ID'
        primary key,
    user_id      bigint                             not null comment '用户ID',
    post_id      bigint                             not null comment '帖子ID',
    read_count   int      default 1                 not null comment '阅读次数',
    last_read_at datetime default CURRENT_TIMESTAMP not null comment '最近阅读时间',
    created_at   datetime default CURRENT_TIMESTAMP not null comment '首次阅读时间',
    updated_at   datetime default CURRENT_TIMESTAMP not null on update CURRENT_TIMESTAMP comment '更新时间',
    constraint uk_post_read
        unique (user_id, post_id)

    comment '帖子阅读记录表';

    on post_reads (last_read_at);

create index idx_post_reads_post
    on post_reads (post_id);

create index idx_post_reads_user
    on post_reads (user_id);

create table post_tags
(
    id         bigint auto_increment comment '关系ID'
        primary key,
    post_id    bigint                             not null comment '帖子ID，逻辑关联 posts.id',
    tag_id     bigint                             not null comment '标签ID，逻辑关联 tags.id',
    created_at datetime default CURRENT_TIMESTAMP not null comment '创建时间',
    constraint uk_post_tag
        unique (post_id, tag_id)
)
    comment '帖子标签关系表';

create index idx_post_tags_post
    on post_tags (post_id);

create index idx_post_tags_tag
    on post_tags (tag_id);

create table posts
(
    id                   bigint auto_increment comment '帖子ID'
        primary key,
    title                varchar(255)                          not null comment '标题',
    description          text                                  null comment '简介/摘要',
    thumbnail_url        text                                  null comment '缩略图URL',
    hero_url             text                                  null comment '头图URL',
    content_md           longtext                              null comment 'Markdown内容',
    content_json         json                                  null comment '富文本JSON内容',
    tag_id               bigint                                not null,
    author_id            bigint                                not null comment '作者ID，逻辑关联 users.id',
    status               varchar(20) default 'normal'          not null comment '状态：normal/deleted/hidden/draft',
    is_hero              tinyint(1)  default 0                 not null comment '是否精选',
    is_pinned_by_author  tinyint(1)  default 0                 not null comment '是否作者置顶',
    is_pinned_by_manager tinyint(1)  default 0                 not null comment '是否管理员置顶',
    num_likes            int         default 0                 not null comment '点赞数',
    num_comments         int         default 0                 not null comment '评论数',
    num_reads            int         default 0                 not null comment '阅读数',
    num_bookmarks        int         default 0                 not null comment '收藏数',
    created_at           datetime    default CURRENT_TIMESTAMP not null comment '创建时间',
    updated_at           datetime    default CURRENT_TIMESTAMP not null on update CURRENT_TIMESTAMP comment '更新时间'
)
    comment '帖子表';

create index idx_posts_author
    on posts (author_id);

create index idx_posts_hero
    on posts (is_hero);

create index idx_posts_hot
    on posts (num_likes, num_comments, num_reads);

create index idx_posts_pinned_author
    on posts (author_id, is_pinned_by_author);

create index idx_posts_status_created
    on posts (status, created_at);

create index idx_posts_tag
    on posts (tag_id);

create table tag_follows
(
    id         bigint auto_increment comment '标签关注ID'
        primary key,
    user_id    bigint                             not null comment '用户ID',
    tag_id     bigint                             not null comment '标签ID',
    created_at datetime default CURRENT_TIMESTAMP not null comment '关注时间',
    constraint uk_tag_follow
        unique (user_id, tag_id)
)
    comment '标签关注表';

create index idx_tag_follows_tag
    on tag_follows (tag_id);

create index idx_tag_follows_user
    on tag_follows (user_id);

create table tags
(
    tag_id            bigint auto_increment comment '标签ID'
        primary key,
    tag_name          varchar(100)                       not null comment '标签名称',
    description       text                               null comment '标签描述',
    num_tag_posts     int      default 0                 not null comment '标签下帖子数',
    num_tag_followers int      default 0                 not null comment '标签关注人数',
    created_at        datetime default CURRENT_TIMESTAMP not null comment '创建时间',
    updated_at        datetime default CURRENT_TIMESTAMP not null on update CURRENT_TIMESTAMP comment '更新时间',
    constraint tag_name
        unique (tag_name)
)
    comment '标签表';

create index idx_tags_hot
    on tags (num_tag_posts, num_tag_followers);

create index idx_tags_name
    on tags (tag_name);

create table users
(
    id              bigint auto_increment comment '用户ID'
        primary key,
    username        varchar(50)                          not null comment '用户名',
    password        varchar(255)                         not null comment '密码',
    bio             text                                 null comment '个人简介',
    sex_type        varchar(20)                          null comment '性别',
    avatar_url      text                                 null comment '头像URL',
    banner_url      text                                 null comment '主页背景图URL',
    location        varchar(100)                         null comment '所在地',
    external_links  json                                 null comment '外部链接',
    is_verified     tinyint(1) default 0                 not null comment '是否认证',
    is_banned       tinyint(1) default 0                 not null comment '是否封禁',
    is_deactivated  tinyint(1) default 0                 not null comment '是否注销/停用',
    ban_reason      varchar(255)                         null comment '封禁原因',
    banned_at       datetime                             null comment '封禁时间',
    banned_until    datetime                             null comment '封禁截止时间',
    followers       int        default 0                 not null comment '粉丝数',
    following       int        default 0                 not null comment '关注数',
    total_posts     int        default 0                 not null comment '发帖数',
    total_views     int        default 0                 not null comment '总浏览量',
    total_likes     int        default 0                 not null comment '总获赞数',
    total_comments  int        default 0                 not null comment '总评论数',
    total_bookmarks int        default 0                 not null comment '总收藏数',
    joined_at       datetime   default CURRENT_TIMESTAMP not null comment '注册时间',
    last_seen_at    datetime                             null comment '最近活跃时间',
    real_name       varchar(50)                          not null comment '真名',
    phone           varchar(20)                          not null comment '电话号码',
    email           varchar(255)                         not null comment '邮箱',
    student_id      varchar(50)                          not null comment '学号',
    created_at      datetime   default CURRENT_TIMESTAMP not null comment '创建时间',
    updated_at      datetime   default CURRENT_TIMESTAMP not null on update CURRENT_TIMESTAMP comment '更新时间',
    constraint username
        unique (username)
)
    comment '用户表';

create index idx_users_status
    on users (is_banned, is_deactivated);

create index idx_users_username
    on users (username);
